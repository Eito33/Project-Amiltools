# Client :

    Calendar :
        -> togglePopup :: Fonction qui permet d'afficher le popup.
        -> getCalendarEvents :: Fonction qui permet d'afficher le calendrier.
        -> addEvent :: Fonction qui permet l'ajout d'un evenement sur le calendrier.
        -> removeEvent :: Fonction qui permet la suppression d'un evenement sur le calendrier.
        -> handleSelectDate :: Fonction qui permet de sélectionner une date.
        -> handleSelectEvent :: Fonction qui permet de sélectionner un événement.
        -> renderContentPopup :: Fonction qui permet le rendu de contenu du Popup.
        -> toggleCB :: Fonction qui permet de gèrer la checkbox.
        -> renderCheckBox :: Fonction qui permet le rendu de la checkbox.
        -> renderPopup :: Fonction qui permet le rendu du Popup.

    PopupCalendar :
        ->  Prends en paramètre des props pour afficher le bon popup calendar.
